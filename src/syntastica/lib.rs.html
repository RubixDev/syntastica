22/// Source code with theme-independent style information attached, as returned by the [`Processor`].
23///
24/// The type is a vector of lines of the source code, and each line is a vector of 2-tuples.
25/// The first element of each 2-tuple is a slice of the source text, and the second
26/// element is an optional theme key for that region of text. The theme key is guaranteed to be
27/// present in [`THEME_KEYS`](theme::THEME_KEYS). The [`render`] function takes this type,
28/// a [`Renderer`], and a [`ResolvedTheme`] as arguments in order to render the text for end users.
29pub type Highlights<'src> = Vec<Vec<(&'src str, Option<&'static str>)>>;

31/// Source code with theme-specific style information attached.
32///
33/// The type is a vector of lines of the source code, and each line is a vector of 2-tuples.
34/// The first element of each 2-tuple is a slice of the source text, and the second
35/// element is an optional style for that region of text.
36pub type ThemedHighlights<'src> = Vec<Vec<(&'src str, Option<Style>)>>;

38/// Convenience function for [processing](Processor) and directly [rendering](render) code once.
39///
40/// **Only use this function if you do not plan to highlight multiple inputs!**
41/// When planning to render the same input multiple times, use [`Processor::process_once`] instead.
42/// When planning to process multiple different inputs with the same [`LanguageSet`], use a
43/// [`Processor`] and call [`Processor::process`] for each input.
44///
45/// # Errors
46///
47/// The function may error in the following situations:
48///
49/// - [`Processor::process_once`] returns an error.
50/// - The `TryInto<ResolvedTheme>` implementation on `T` errors. If `T` is [`Theme`](theme::Theme),
51///   look at [`Theme::resolve_links`](theme::Theme::resolve_links) for when this might happen.
52pub fn highlight<'s, S, T, E>(
53    code: impl AsRef<str>,
54    language: impl Into<S::Language>,
55    language_set: &'s S,
56    renderer: &mut impl Renderer,
57    theme: T,
58) -> Result<String>
59where
60    S: LanguageSet<'s>,
61    T: TryInto<ResolvedTheme, Error = E>,
62    crate::Error: From<E>,
63{
64    Ok(render(
65        &Processor::process_once(code.as_ref(), language, language_set)?,
66        renderer,
67        theme.try_into()?,
68    ))
69}
