16#[proc_macro]
17pub fn parsers_git(_: TokenStream) -> TokenStream {
18    let mut dedup_ffi_funcs = LANGUAGE_CONFIG.languages.clone();
19    dedup_ffi_funcs.sort_unstable_by_key(|lang| lang.parser.ffi_func.clone());
20    dedup_ffi_funcs.dedup_by_key(|lang| lang.parser.ffi_func.clone());
21    dedup_ffi_funcs
22        .iter()
23        .map(|lang| {
24            let name = &lang.name;
25            let url = &lang.parser.git.url;
26            let rev = &lang.parser.git.rev;
27            let external_c = lang.parser.external_scanner.c;
28            let external_cpp = lang.parser.external_scanner.cpp;
29            let path = match &lang.parser.git.path {
30                Some(path) => quote_use! { Some(#path) },
31                None => quote_use! { None },
32            };
33            let wasm = lang.wasm;
34            let wasm_unknown = lang.wasm_unknown;
35            let generate = lang.parser.generate;
36            quote! {
37                #[cfg(feature = #name)]
38                compile_parser(#name, #url, #rev, #external_c, #external_cpp, #path, #wasm, #wasm_unknown, #generate)?;
39            }
40        })
41        .collect::<proc_macro2::TokenStream>()
42        .into()
43}

45fn not_wasm_cfg(lang: &Language) -> proc_macro2::TokenStream {
46    let raw_wasm_cfg = quote! { target_family = "wasm" };
47    let raw_wasm_unknown_cfg = quote! { all(target_arch = "wasm32", target_vendor = "unknown", target_os = "unknown", target_env = "") };
48    match (
49        lang.wasm,
50        lang.parser.external_scanner.cpp || !lang.wasm_unknown,
51    ) {
52        (false, _) => quote! { , not(#raw_wasm_cfg) },
53        (_, true) => quote! { , not(#raw_wasm_unknown_cfg) },
54        _ => quote! {},
55    }
56}

58#[proc_macro]
59pub fn parsers_ffi(_: TokenStream) -> TokenStream {
60    let mut dedup_ffi_funcs = LANGUAGE_CONFIG.languages.clone();
61    dedup_ffi_funcs.sort_unstable_by_key(|lang| lang.parser.ffi_func.clone());
62    dedup_ffi_funcs.dedup_by_key(|lang| lang.parser.ffi_func.clone());
63    let extern_c = dedup_ffi_funcs.iter().map(|lang| {
64        let name_str = &lang.name;
65        let ffi_func = format_ident!("{}", lang.parser.ffi_func);
66        let not_wasm_cfg = not_wasm_cfg(lang);
67        quote! {
68            #[cfg(all(feature = #name_str #not_wasm_cfg))]
69            fn #ffi_func() -> ::syntastica_core::language_set::Language;
70        }
71    });
72    let functions = LANGUAGE_CONFIG.languages.iter().map(|lang| {
73        let feat = lang.group.to_string();
74        let name = format_ident!("{}", lang.name);
75        let name_str = &lang.name;
76        let ffi_func = format_ident!("{}", lang.parser.ffi_func);
77        let doc = format!(
78            "Get the parser for [{}]({}/tree/{}). {}",
79            lang.name,
80            lang.parser.git.url,
81            lang.parser.git.rev,
82            match (
83                lang.wasm,
84                lang.parser.external_scanner.cpp || !lang.wasm_unknown
85            ) {
86                (false, _) => "(not supported on WebAssembly targets)",
87                (_, true) => "(not supported on the `wasm32-unknown-unknown` target)",
88                _ => "",
89            },
90        );
91        // disable unsupported parsers on wasm
92        let not_wasm_cfg = not_wasm_cfg(lang);
93        quote! {
94            #[cfg(all(any(feature = #feat, feature = #name_str) #not_wasm_cfg))]
95            #[doc = #doc]
96            pub fn #name() -> ::syntastica_core::language_set::Language {
97                #[cfg(not(all(feature = "docs", doc)))]
98                unsafe { #ffi_func() }
99                #[cfg(all(feature = "docs", doc))]
100                ::std::unimplemented!()
101            }
102        }
103    });
104    parsers(
105        "syntastica_parsers_git",
106        functions,
107        |_| true,
108        Some(quote! {
109            #[cfg(not(all(feature = "docs", doc)))]
110            extern "C" {
111                #(#extern_c)*
112            }
113        }),
114        "",
115    )
116}

128fn parsers_rust(crate_name: &str, crates_io: bool, query_suffix: &str) -> TokenStream {
129    let functions = LANGUAGE_CONFIG.languages.iter().map(|lang| {
130        let feat = lang.group.to_string();
131        let name = format_ident!("{}", lang.name);
132        let name_str = &lang.name;
133        let (doc, body) = match &lang.parser.rust_const {
134            Some(ident) if (!crates_io && !lang.parser.generate) || lang.parser.crates_io.is_some() => {
135                let ident = format_ident!("{ident}");
136                let package = format_ident!("{}", lang.parser.package.replace('-', "_"));
137                (
138                    format!(
139                        "Get the parser for [{}]({}/tree/{}).",
140                        lang.name, lang.parser.git.url, lang.parser.git.rev,
141                    ),
142                    quote! { ::syntastica_core::language_set::Language::new(#package::#ident) }
143                )
144            },
145            _ => (
146                "**This parser is not supported by this parser collection and thus this function will panic!**"
147                    .to_owned(),
148                quote! { ::std::unimplemented!() }
149            ),
150        };
151        quote! {
152            #[cfg(any(feature = #feat, feature = #name_str))]
153            #[doc = #doc]
154            pub fn #name() -> ::syntastica_core::language_set::Language {
155                #body
156            }
157        }
158    });
159    parsers(
160        crate_name,
161        functions,
162        |lang| {
163            lang.parser.rust_const.is_some()
164                && ((!crates_io && !lang.parser.generate) || lang.parser.crates_io.is_some())
165        },
166        None,
167        query_suffix,
168    )
169}

171fn parsers(
172    crate_name: &str,
173    functions: impl Iterator<Item = proc_macro2::TokenStream>,
174    filter: impl Fn(&&Language) -> bool,
175    extra: Option<proc_macro2::TokenStream>,
176    query_suffix: &str,
177) -> TokenStream {
178    let list = LANGUAGE_CONFIG
179        .languages
180        .iter()
181        .filter(&filter)
182        .map(|lang| {
183            let name_str = &lang.name;
184            let variant = format_ident!("{}", lang.name.to_pascal_case());
185            let not_wasm_cfg = not_wasm_cfg(lang);
186            quote! { #[cfg(all(feature = #name_str #not_wasm_cfg))] Lang::#variant }
187        });
188    let names_list = LANGUAGE_CONFIG
189        .languages
190        .iter()
191        .filter(&filter)
192        .map(|lang| {
193            let name_str = &lang.name;
194            let not_wasm_cfg = not_wasm_cfg(lang);
195            quote! { #[cfg(all(feature = #name_str #not_wasm_cfg))] #name_str }
196        });
197    let file_types = LANGUAGE_CONFIG
198        .languages
199        .iter()
200        .filter(&filter)
201        .flat_map(|lang| {
202            let name_str = &lang.name;
203            let variant = format_ident!("{}", lang.name.to_pascal_case());
204            let not_wasm_cfg = not_wasm_cfg(lang);
205            lang.file_types.iter().map(move |ft| {
206                let ft = format_ident!("{ft:?}");
207                quote! {
208                    #[cfg(all(feature = #name_str #not_wasm_cfg))]
209                    (::syntastica_core::language_set::FileType::#ft, Lang::#variant)
210                }
211            })
212        });
213    let mut langs_sorted_by_group = LANGUAGE_CONFIG.languages.clone();
214    langs_sorted_by_group.sort_by_key(|lang| lang.group);
215    let func_map = langs_sorted_by_group.iter().filter(&filter).map(|lang| {
216        let name_str = &lang.name;
217        let variant = format_ident!("{}", lang.name.to_pascal_case());
218        let not_wasm_cfg = not_wasm_cfg(lang);
219        quote! {
220            #[cfg(all(feature = #name_str #not_wasm_cfg))]
221            {
222                _map.insert(Lang::#variant, _idx);
223                _idx += 1;
224            }
225        }
226    });
227    let funcs = langs_sorted_by_group.iter().filter(&filter).map(|lang| {
228        let name = format_ident!("{}", lang.name);
229        let name_str = &lang.name;
230        let not_wasm_cfg = not_wasm_cfg(lang);
231        quote! { #[cfg(all(feature = #name_str #not_wasm_cfg))] #name }
232    });
233    let queries = langs_sorted_by_group.iter().filter(&filter).map(|lang| {
234        let name_str = &lang.name;
235        let highlights = format_ident!("{}_HIGHLIGHTS{query_suffix}", lang.name.to_uppercase());
236        let injections = format_ident!("{}_INJECTIONS{query_suffix}", lang.name.to_uppercase());
237        let locals = format_ident!("{}_LOCALS{query_suffix}", lang.name.to_uppercase());
238        let not_wasm_cfg = not_wasm_cfg(lang);
239
240        quote! { #[cfg(all(feature = #name_str #not_wasm_cfg))] [
241            ::syntastica_queries::#highlights,
242            ::syntastica_queries::#injections,
243            ::syntastica_queries::#locals,
244        ] }
245    });
246    let lang_enum_example_use = format!("use {crate_name}::{{Lang, LANGUAGES, LANGUAGE_NAMES}};");
247    let lang_enum = LANGUAGE_CONFIG
248        .languages
249        .iter()
250        .filter(&filter)
251        .map(|lang| {
252            let feat = lang.group.to_string();
253            let name_str = &lang.name;
254            let ft_support = if lang.file_types.is_empty() {
255                Cow::Borrowed("supports no file types")
256            } else {
257                format!(
258                    "supports these file types: {}",
259                    lang.file_types
260                        .iter()
261                        .map(|ft| format!(
262                            "[`{ft}`](::syntastica_core::language_set::FileType::{ft:?})"
263                        ))
264                        .collect::<Vec<_>>()
265                        .join(", ")
266                )
267                .into()
268            };
269            let doc = format!("Provides the [`{name_str}`] language, {ft_support}.");
270            let variant = format_ident!("{}", lang.name.to_pascal_case());
271            let not_wasm_cfg = not_wasm_cfg(lang);
272            quote! {
273                #[doc = #doc]
274                #[cfg(all(any(feature = #feat, feature = #name_str) #not_wasm_cfg))]
275                #variant
276            }
277        });
278    let lang_get_match = LANGUAGE_CONFIG
279        .languages
280        .iter()
281        .filter(&filter)
282        .map(|lang| {
283            let name = format_ident!("{}", lang.name);
284            let name_str = &lang.name;
285            let variant = format_ident!("{}", lang.name.to_pascal_case());
286            let not_wasm_cfg = not_wasm_cfg(lang);
287            quote! {
288                #[cfg(all(feature = #name_str #not_wasm_cfg))]
289                Self::#variant => #name(),
290            }
291        });
292    let lang_set_type = quote! { type Language = Lang; };
293    let cfg_test = quote! { #[cfg(test)] };
294    let lang_tests = LANGUAGE_CONFIG.languages.iter().filter(&filter).map(|lang| {
295        let name = format_ident!("{}", lang.name);
296        let name_str = &lang.name;
297        let variant = format_ident!("{}", lang.name.to_pascal_case());
298        let not_wasm_cfg = not_wasm_cfg(lang);
299        quote! {
300            #[test]
301            #[cfg(all(feature = #name_str #not_wasm_cfg))]
302            fn #name() {
303                assert_eq!(crate::#name(), crate::Lang::#variant.get());
304                assert!(::syntastica_core::language_set::LanguageSet::get_language(&crate::LanguageSetImpl::new(), crate::Lang::#variant).is_ok());
305            }
306        }
307    });
308
309    quote_use! {
310        # use std::{borrow::Cow, collections::HashMap};
311
312        # use syntastica_core::{
313            language_set::{HighlightConfiguration, LanguageSet, Language, FileType, SupportedLanguage},
314            Error, Result,
315            theme::THEME_KEYS,
316        };
317        # use once_cell::sync::{Lazy, OnceCell};
318
319        #extra
320
321        /// A list of all languages supported by the current feature set.
322        pub const LANGUAGES: &[Lang] = &[#(#list),*];
323        const LANG_COUNT: usize = LANGUAGES.len();
324
325        /// A list of all language names supported by the current feature set.
326        pub const LANGUAGE_NAMES: &[&str] = &[#(#names_list),*];
327
328        #(#functions)*
329
330        // TODO: use "perfect" hashmap with compile-time known keys
331        static FILE_TYPE_MAP: Lazy<HashMap<FileType, Lang>>
332            = Lazy::new(|| HashMap::from([#(#file_types),*]));
333
334        // TODO: use "perfect" hashmap with compile-time known keys
335        static IDX_MAP: Lazy<HashMap<Lang, usize>> = Lazy::new(|| {
336            let mut _map = HashMap::new();
337            let mut _idx = 0;
338            #(#func_map)*
339            _map
340        });
341
342        const QUERIES: &[[&str; 3]] = &[#(#queries),*];
343        const FUNCS: &[fn() -> Language] = &[#(#funcs),*];
344
345        fn __get_language(idx: usize) -> Result<HighlightConfiguration> {
346            let lang = FUNCS[idx]();
347            let mut conf = HighlightConfiguration::new(
348                lang,
349                LANGUAGE_NAMES[idx],
350                QUERIES[idx][0],
351                QUERIES[idx][1],
352                QUERIES[idx][2],
353            )?;
354            conf.configure(THEME_KEYS);
355            Ok(conf)
356        }
357
358        /// An enum of every supported language in the current feature set.
359        ///
360        /// An instance of the respective tree-stter
361        /// [`Language`](::syntastica_core::language_set::Language) can be obtained with the
362        /// [`get`](Lang::get) method.
363        ///
364        /// You can also get a [`Lang`] from its name using
365        /// [`for_name`](::syntastica_core::language_set::SupportedLanguage::for_name), or for a
366        /// [`FileType`](::syntastica_core::language_set::FileType) using
367        /// [`for_file_type`](::syntastica_core::language_set::SupportedLanguage::for_file_type).
368        /// See the docs for each variant to see its "name" and the supported file types.
369        /// Both of these require the
370        /// [`SupportedLanguage`](::syntastica_core::language_set::SupportedLanguage) trait to be
371        /// in scope.
372        ///
373        /// See [`LANGUAGES`] for a list containing all variants and [`LANGUAGE_NAMES`] for a list
374        /// of all valid names.
375        ///
376        /// The enum is marked as non-exhaustive for two reasons:
377        ///
378        /// 1. New languages may be added in the future
379        /// 2. The variants are enabled/disabled by features
380        ///
381        /// # Example
382        ///
383        /// ```
384        #[doc = #lang_enum_example_use]
385        /// use syntastica_core::language_set::{SupportedLanguage, FileType};
386        ///
387        /// // you can get a `Lang` from its name
388        /// assert_eq!(Lang::Rust, Lang::for_name("rust", &()).unwrap());
389        /// // and for a file type
390        /// assert_eq!(Some(Lang::Rust), Lang::for_file_type(FileType::Rust, &()));
391        ///
392        /// // `LANGUAGES` is a list of all variants,
393        /// // `LANGUAGE_NAMES` is a list of all variant names
394        /// for (&lang, &name) in LANGUAGES.iter().zip(LANGUAGE_NAMES) {
395        ///     assert_eq!(lang, Lang::for_name(name, &()).unwrap());
396        ///
397        ///     // `Lang` instances can be turned into strings
398        ///     assert_eq!(lang, Lang::for_name(<Lang as SupportedLanguage<'_, ()>>::name(&lang), &()).unwrap());
399        ///     assert_eq!(lang, Lang::for_name(lang.to_string(), &()).unwrap());
400        ///     assert_eq!(lang, Lang::for_name(lang.as_ref(), &()).unwrap());
401        ///     let lang_name: &'static str = lang.into();
402        ///     assert_eq!(lang, Lang::for_name(lang_name, &()).unwrap());
403        /// }
404        /// ```
405        #[non_exhaustive]
406        #[derive(
407            Debug,
408            Clone,
409            Copy,
410            PartialEq,
411            Eq,
412            Hash,
413            ::strum::Display,
414            ::strum::AsRefStr,
415            ::strum::IntoStaticStr,
416            ::strum::EnumString,
417        )]
418        #[strum(serialize_all = "snake_case", use_phf)]
419        pub enum Lang {
420            #(#lang_enum),*
421        }
422
423        impl Lang {
424            /// Get an instance of the respective
425            /// [`Language`](::syntastica_core::language_set::Language).
426            pub fn get(&self) -> Language {
427                match self {
428                    #(#lang_get_match)*
429                    _ => unreachable!("all variants are matched")
430                }
431            }
432        }
433
434        impl<S> SupportedLanguage<'_, S> for Lang {
435            fn name(&self) -> Cow<'_, str> {
436                Cow::Borrowed(self.into())
437            }
438
439            fn for_name(name: impl AsRef<str>, _set: &S) -> Result<Self> {
440                <Self as ::std::str::FromStr>::from_str(name.as_ref())
441                    .map_err(|_| Error::UnsupportedLanguage(name.as_ref().to_owned()))
442            }
443
444            fn for_file_type(file_type: FileType, _set: &S) -> Option<Self> {
445                FILE_TYPE_MAP
446                    .get(&file_type)
447                    .map(|name| (*name).into())
448            }
449        }
450
451        /// An implementation of [`LanguageSet`](::syntastica_core::language_set::LanguageSet)
452        /// including all languages in the enabled feature set.
453        ///
454        /// Languages are loaded the first time they are requested and will then be reused for
455        /// later accesses. To pre-load a list of languages, use
456        /// [`preload`](LanguageSetImpl::preload) or [`preload_all`](LanguageSetImpl::preload_all).
457        pub struct LanguageSetImpl([OnceCell<HighlightConfiguration>; LANG_COUNT]);
458
459        impl LanguageSet<'_> for LanguageSetImpl {
460            #lang_set_type
461
462            fn get_language(&self, language: Self::Language) -> Result<&HighlightConfiguration> {
463                let idx = IDX_MAP[&language];
464                self.0[idx].get_or_try_init(|| __get_language(idx))
465            }
466        }
467
468        impl LanguageSetImpl {
469            /// Create a new [`LanguageSetImpl`] with no pre-loaded languages.
470            pub fn new() -> Self {
471                #[allow(clippy::declare_interior_mutable_const)]
472                const INIT: OnceCell<HighlightConfiguration> = OnceCell::new();
473                Self([INIT; LANG_COUNT])
474            }
475
476            /// Pre-load the given list of languages.
477            ///
478            /// To pre-load all supported languages, use [`preload_all`](LanguageSetImpl::preload_all).
479            pub fn preload(&self, languages: &[Lang]) -> Result<()> {
480                for lang in languages {
481                    let idx = IDX_MAP[lang];
482                    let entry = &self.0[idx];
483                    if entry.get().is_none() {
484                        drop(entry.set(__get_language(idx)?));
485                    }
486                }
487                Ok(())
488            }
489
490            /// Pre-load all languages in this set.
491            ///
492            /// To pre-load a specific set of languages, use [`preload`](LanguageSetImpl::preload).
493            pub fn preload_all(&mut self) -> Result<()> {
494                self.preload(LANGUAGES)
495            }
496        }
497
498        impl Default for LanguageSetImpl {
499            fn default() -> Self {
500                Self::new()
501            }
502        }
503
504        #cfg_test
505        mod tests {
506            #(#lang_tests)*
507        }
508    }
509    .into()
510}
